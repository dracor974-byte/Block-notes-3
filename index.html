import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js"; import { getDatabase, ref, push, onValue } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-database.js"; /* 🔥 FIREBASE CONFIG */ const firebaseConfig = { apiKey: "XXX", authDomain: "XXX", databaseURL: "XXX", projectId: "XXX", appId: "XXX" }; const app = initializeApp(firebaseConfig); const db = getDatabase(app); const notesRef = ref(db, "chat_notes"); /* STATE */ let currentUser = null; let currentRoom = "serie"; let cloudNotes = []; /* LOGIN */ window.login = function(id,name,color,emoji){ currentUser = {id,name,color,emoji}; document.getElementById("login").style.display="none"; document.getElementById("app").style.display="block"; Notification.requestPermission(); }; /* SWITCH ROOM */ window.switchRoom = function(room,el){ currentRoom = room; document.querySelectorAll(".channel").forEach(c=>c.classList.remove("active")); el.classList.add("active"); render(); }; /* SEND MESSAGE (FIREBASE + IMAGE + TEXT) */ window.postNote = function(){ const text = document.getElementById("note-input").value.trim(); if(!text && !window.pendingImageUrl) return; push(notesRef,{ content: text, imageUrl: window.pendingImageUrl || null, authorId: currentUser.id, authorName: currentUser.name, color: currentUser.color, room: currentRoom, createdAt: Date.now() }); document.getElementById("note-input").value=""; window.pendingImageUrl=null; }; /* REALTIME SYNC */ onValue(notesRef,(snap)=>{ const data = snap.val(); if(!data) return; cloudNotes = Object.values(data); render(); cloudNotes.forEach(n=>{ if(n.authorId !== currentUser?.id){ notify(n); } }); }); /* RENDER */ function render(){ const list = document.getElementById("notes-list"); const notes = cloudNotes .filter(n=>n.room===currentRoom) .sort((a,b)=>b.createdAt-a.createdAt); list.innerHTML = notes.map(n=>`
${n.authorName}
${n.content || ""}
${n.imageUrl ? `` : ""}
`).join(""); } /* VOICE */ window.speak = function(text){ if(!text) return; speechSynthesis.cancel(); const u = new SpeechSynthesisUtterance(text); u.lang = "fr-FR"; u.rate = 0.95; u.pitch = 1.1; speechSynthesis.speak(u); }; /* NOTIFICATIONS */ function notify(note){ if(Notification.permission !== "granted") return; if(!currentUser) return; if(note.authorId === currentUser.id) return; new Notification("📝 " + note.authorName,{ body: note.content || "image", }); }
